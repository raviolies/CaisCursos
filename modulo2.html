Fatorial de um número

n = int(input("Entre com um número")) //Começamos declarando a variável N que armazena o valor que será lido
fat = 1 //Depois, criamos a variável que armazenará o resultado do fatorial
for i in range(1, n+1): //Temos um for
fat = fat * i //Aqui temos uma operação para realizar o cálculo
print(fat) //E para finalizar, imprimimos o resultado.


Número primo

n = int(input("Entre com um número") //Começamos declarando a variável N que armazena o valor que será lido
x=0 //Criamos e inicializamos a variável X, e atribuímos o valor 0 para que ela não acumule um lixo de memória
for i in range(1,n+1): //Temos um for
if(n%i==0): //Aqui há um if para verificar se o resto da operação n dividido por i é igual a zero
x=x+1 //Então acrescentamos 1 unidade ao valor de x a cada vez que o loop é executado.
if x==2: //Temos um if que verifica se o valor de x é igual a 2 pois se um número é primo, ele só poderá ser dividido por 1 ou por ele mesmo
print("Número é primo") //Imprimindo
else: //Se não,
print("Número não é primo")//Imprimindo


Número de Euler

n = int(input("Entre com um número")) //Começamos declarando a variável N que armazena o valor que será lido
E=1 //Depois declaramos a variável E que armazena o número de Euler, que é uma constante e não pode ser alterado
for i in range(1, n+1):
fat=1
for j in range(1,i+1):
fat= fat*j //Aqui temos a operação para que consigamos imprimir a constante
E= E+(1/fat) //Aqui também
print(f"O número de euler de {n} é {E}")


Vetores

n = int(input("Entre com a dimensão do vetor: "))
vetor=[] //Utilizamos estas chaves para indicar a posição do vetor
for i in range(0, n):
vetor.append(int(input(f"Entre com a posição {i} do vetor: "))) //Usamos esta função para trabalhar com vetores
for i in range(0, n):
print(f"[{vetor[i]}]",end=' ')


Matriz

linhas = int(input("Digite o número de linhas: "))
colunas = int(input("Digite o número de colunas: "))
matriz = np.zeros((linhas, colunas)) //Aqui inserimos o n° das linhas e das colunas
for i in range(linhas):
for j in range(colunas):
valor = int(input(f"Digite o valor para a posição [{i}][{j}]: "))
matriz[i, j] = valor
for i in range(linhas):
print()
for j in range(colunas):
print(f"[{matriz[i,j]}]",end=' ')
