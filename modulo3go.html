FATORIAL

package main

import ( "fmt")

func fatorial(n int) int {

if n == 0 { return 1 } return n * fatorial(n-1) }

func main() {

var num int

fmt.Print("Digite um número: ")

fmt.Scan(&num)

fmt.Printf("Fatorial de %d é %d\n", num, fatorial(num))}


PRIMO

package main

import ("fmt")

func ehPrimo(n int) bool {

if n <= 1 {

return false }

for i := 2; i*i <= n; i++ {

if n%i == 0 {

return false } }

return true }


func main() {

var num int

fmt.Print("Digite um número: ")

fmt.Scan(&num)

if ehPrimo(num) {

fmt.Printf("%d é primo.\n", num)

} else {

fmt.Printf("%d não é primo.\n", num) } }


EULER

package main

import ("fmt")

func calcularEuler(termos int) float64 {

e := 1.0

fatorial := 1.0

for i := 1; i <= termos; i++ {

fatorial *= float64(i)

e += 1 / fatorial }

return e }

ARQUIVOS

CRIAR ARQUIVO

package main

import ("fmt"

"os")

func main() {

arquivo, err := os.Create("exemplo.txt")

if err != nil {

fmt.Println(err)

return}

defer arquivo.Close()

arquivo.WriteString("Este é um arquivo de exemplo.\n")

fmt.Println("Arquivo criado com sucesso.") }

EDITAR ARQUIVO

package main

import (

"fmt"

"os")

func main() {

arquivo, err := os.OpenFile("exemplo.txt", os.O_APPEND|os.O_WRONLY, 0644)

if err != nil {

fmt.Println(err)

return }

defer arquivo.Close()

_, err = arquivo.WriteString("Adicionando uma nova linha.\n")

if err != nil {

fmt.Println(err)

return }

fmt.Println("Arquivo editado com sucesso.") }
