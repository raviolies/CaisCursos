BUSCA BINÁRIA

package main

import "fmt"

func buscaBinaria(arr []int, x int) int {

esq, dir := 0, len(arr)-1

for esq <= dir {

meio := esq + (dir-esq)/2

if arr[meio] == x {

return meio}

if arr[meio] < x {

esq = meio + 1}

else{

dir = meio - 1}}

return -1}

func main() {

arr := []int{1, 3, 5, 7, 9, 11}

x := 7

fmt.Println("Posição:", buscaBinaria(arr, x))}

ORDENAÇÃO BOLHA

package main

import "fmt"

func ordenacaoBolha(arr []int) {

n := len(arr)

for i := 0; i < n-1; i++ {

for j := 0; j < n-i-1; j++ {

if arr[j] > arr[j+1] {

arr[j], arr[j+1] = arr[j+1], arr[j] }

} } }

func main() {

arr := []int{5, 1, 4, 2, 8}

ordenacaoBolha(arr)

fmt.Println("Ordenado:", arr)}


SELEÇÃO DIRETA

package main

import "fmt"

func ordenacaoSelecao(arr []int) {

n := len(arr)

for i := 0; i < n-1; i++ {

min := i

for j := i + 1; j < n; j++ {

if arr[j] < arr[min] {

min = j }}

arr[i], arr[min] = arr[min], arr[i] } }

func main() {

arr := []int{29, 10, 14, 37, 13}

ordenacaoSelecao(arr)

fmt.Println("Ordenado:", arr)}


INSERÇÃO DIRETA

package main

import "fmt"

func ordenacaoInsercao(arr []int) {

n := len(arr)

for i := 1; i < n; i++ {

chave := arr[i]

j := i - 1

for j >= 0 && arr[j] > chave {

arr[j+1] = arr[j]

j--}

arr[j+1] = chave} }

func main() {

arr := []int{12, 11, 13, 5, 6}

ordenacaoInsercao(arr)

fmt.Println("Ordenado:", arr)}


INSERÇÃO BINÁRIA

package main

import "fmt"

func insercaoBinaria(arr []int) {

n := len(arr)

for i := 1; i < n; i++ {

x := arr[i]

esq, dir := 0, i

for esq < dir {

meio := (esq + dir) / 2

if arr[meio] <= x {

esq = meio + 1 }

else { dir = meio } }

for j := i; j > esq; j-- {

arr[j] = arr[j-1] }

arr[esq] = x } }

func main() {

arr := []int{37, 23, 0, 17, 12, 72, 31}

insercaoBinaria(arr)

fmt.Println("Ordenado:", arr)}
